AJAX 如何調用？

  • Get 請求

    
$.ajax({
    url: 'http://localhost:8080/api/book',
    method: 'get',
    success: function (data) {
        console.log(data);
    },
    error: function (e) {
        console.log(e);
    }
});

  • Get 請求未加 Header

    
$.ajax({
    url: 'http://localhost:8080/api/secured/book',
    method: 'get',
    success: function (data) {
        console.log(data);
    },
    error: function (e) {
        console.log(e);
    }
});

  • Get 請求加非法 Header

    
$.ajax({
    url: 'http://localhost:8080/api/secured/book',
    method: 'get',
    headers: {
        "Token": 123
    },
    success: function (data) {
        console.log(data);
    },
    error: function (e) {
        console.log(e);
    }
});

  • 同步請求：先取得 Token，再將 Token 放到 Header 中，並發出 GET 請求

    
let token = '';

$.ajax({
    url: 'http://localhost:8080/api/token',
    method: 'get',
    headers: {},
    async: false,
    success: function (data) {
        token = data;
        console.log(data);
    },
    error: function (e) { console.log(e); }
});

$.ajax({
    url: 'http://localhost:8080/api/secured/book',
    method: 'get',
    headers: { "Token": token },
    success: function (data) { console.log(data); },
    error: function (e) { console.log(e); }
});

  • 同步請求：先取得 Token，再將 Token 放到 Header 中，並發出 POST 請求

    
let token = '';

$.ajax({
    url: 'http://localhost:8080/api/token',
    method: 'get',
    headers: {},
    async: false,
    success: function (data) {
        token = data;
        console.log(data);
    },
    error: function (e) { console.log(e); }
});

$.ajax({
    url: 'http://localhost:8080/api/secured/book',
    method: 'post',
    headers: {
        'Token': token,
    },
    dataType: "json",
    data: JSON.stringify (
        {
            name: "HTML、CSS、JS 介紹",
            price: 300
        }
    ),
    contentType: 'application/json',
    success: function (data) { console.log(data); },
    error: function (e) { console.log(e); }
});
